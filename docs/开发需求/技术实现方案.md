## 一、技术架构选型

### 1.1 技术栈选择

| 层级 | 技术选型 | 说明 |
|------|---------|------|
| **桌面框架** | Tauri v2 | Rust后端 + Web前端，轻量高性能 |
| **前端框架** | React 18 + TypeScript | 组件化开发，类型安全 |
| **UI组件库** | shadcn/ui + Tailwind CSS | 可定制性强，轻量，基于Radix UI |
| **后端语言** | Rust | 高性能文件处理，内存安全 |
| **OCR引擎** | PaddleOCR (本地) | 中文识别率高，免费离线使用 |
| **PDF处理** | pdf2image + PyMuPDF | PDF转图片、文本提取 |
| **Excel处理** | calamine (Rust) | 快速读取Excel，无依赖 |
| **数据存储** | SQLite (本地) | 轻量，无需服务器 |
| **打印功能** | 系统原生打印API | Windows打印队列 |

### 4.2 文本规范化服务

```rust
// src-tauri/src/services/normalize_service.rs

use crate::models::config::{BracketStyle, OCRConfig};

pub struct TextNormalizer;

impl TextNormalizer {
    /// 预设OCR纠错词库
    pub fn get_preset_corrections() -> Vec<(&'static str, &'static str)> {
        vec![
            ("责行", "责令"),
            ("公积全", "公积金"),
            ("铁胳", "铁路"),
            ("运翰", "运输"),
            ("法皖", "法院"),
            ("裁走", "裁定"),
            ("准子", "准予"),
            ("强割执行", "强制执行"),
            ("决走", "决定"),
            ("审走", "审定"),
            ("查字", "查字"),
            ("穗字", "穗字"),
            ("天字", "天字"),
            ("越字", "越字"),
        ]
    }
    
    /// 统一括号格式
    pub fn normalize_brackets(text: &str, style: &BracketStyle) -> String {
        match style {
            BracketStyle::Chinese => {
                text.replace('(', "（").replace(')', "）")
            }
            BracketStyle::English => {
                text.replace('（', "(").replace('）', ")")
            }
        }
    }
    
    /// 去除多余空格和换行
    pub fn remove_extra_whitespace(text: &str) -> String {
        text.split_whitespace().collect::<Vec<_>>().join("")
    }
    
    /// 修正常见OCR错误
    pub fn fix_common_errors(text: &str, config: &OCRConfig) -> String {
        let mut result = text.to_string();
        
        // 应用预设纠错
        if config.enable_preset_corrections {
            for (wrong, correct) in Self::get_preset_corrections() {
                result = result.replace(wrong, correct);
            }
        }
        
        // 应用自定义纠错（优先级更高）
        for (wrong, correct) in &config.custom_corrections {
            result = result.replace(wrong, correct);
        }
        
        result
    }
    
    /// 完整规范化流程
    pub fn normalize(text: &str, config: &OCRConfig) -> String {
        let text = Self::remove_extra_whitespace(text);
        let text = Self::fix_common_errors(&text, config);
        Self::normalize_brackets(&text, &config.bracket_style)
    }
}
```

```rust
            }
        }

    }

        for pattern in &patterns {
            if let Ok(re) = Regex::new(pattern) {
                if let Some(caps) = re.captures(text) {
                    return Some(caps.get(1)?.as_str().trim().to_string());
                }
            }
        }
        None
    }
    
    /// 提取执行时间（准予执行日期）
    pub fn extract_execution_date(text: &str) -> Option<String> {
        // 优先查找"准予执行"附近的日期
        let pattern = r"准予执行.*?((\d{4})[年/-](\d{1,2})[月/-](\d{1,2})[日]?))";
        if let Ok(re) = Regex::new(pattern) {
            if let Some(caps) = re.captures(text) {
                let year = caps.get(2)?.as_str();
                let month = caps.get(3)?.as_str();
                let day = caps.get(4)?.as_str();
                return Some(format!("{}年{}月{}日", year, month, day));
            }
        }
        
        // 备选：查找裁定日期
        let pattern2 = r"裁定日期[：:]\s*((\d{4})[年/-](\d{1,2})[月/-](\d{1,2})[日]?)";
        if let Ok(re) = Regex::new(pattern2) {
            if let Some(caps) = re.captures(text) {
                let year = caps.get(2)?.as_str();
                let month = caps.get(3)?.as_str();
                let day = caps.get(4)?.as_str();
                return Some(format!("{}年{}月{}日", year, month, day));
            }
        }
        
        // 最后尝试：任意日期格式
        let pattern3 = r"((\d{4})[年/-](\d{1,2})[月/-](\d{1,2})[日]?)";
        if let Ok(re) = Regex::new(pattern3) {
            if let Some(caps) = re.captures(text) {
                let year = caps.get(2)?.as_str();
                let month = caps.get(3)?.as_str();
                let day = caps.get(4)?.as_str();
                return Some(format!("{}年{}月{}日", year, month, day));
            }
        }
        None
    }
    
    // ============ 辅助方法 ============
    
    fn extract_with_patterns(text: &str, patterns: &[&str]) -> Option<String> {
        for pattern in patterns {
            if let Ok(re) = Regex::new(pattern) {
                if let Some(caps) = re.captures(text) {
                    if let Some(matched) = caps.get(1) {
                        let name = matched.as_str().trim();
                        return Some(Self::clean_company_name(name));
                    }
                }
            }
        }
        None
    }

        } else {
            format!("{}公司", name)
        }
```

```rust
// src-tauri/src/services/match_service.rs

use strsim::jaro_winkler;
use crate::models::config::MatchConfig;

pub struct FuzzyMatcher {
    config: MatchConfig,
}

impl FuzzyMatcher {
    pub fn new(config: MatchConfig) -> Self {
        Self { config }
    }
    
    /// 匹配责令号
    pub fn match_order_number(&self, ocr_number: &str, table_numbers: &[String]) -> Option<(String, f64)> {
        let threshold = self.config.get_threshold();
        let mut best_match: Option<(String, f64)> = None;
        
        for table_number in table_numbers {
            // 先尝试精确匹配
            if ocr_number == table_number {
                return Some((table_number.clone(), 1.0));
            }
            
            // 模糊匹配
            let similarity = jaro_winkler(ocr_number, table_number);
            
            if similarity >= threshold {
                if best_match.is_none() || similarity > best_match.as_ref().unwrap().1 {
                    best_match = Some((table_number.clone(), similarity));
                }
            }
        }
        
        best_match
    }
    
    /// 匹配文件名
    pub fn match_file_name(&self, keyword: &str, file_name: &str) -> bool {
        let threshold = self.config.get_threshold();
        
        // 包含匹配
        if file_name.contains(keyword) {
            return true;
        }
        
        // 模糊匹配
        let similarity = jaro_winkler(keyword, file_name);
        similarity >= threshold
    }
    
    /// 批量匹配
    pub fn batch_match<T: AsRef<str>>(
        &self,
        ocr_items: &[String],
        table_items: &[T],
    ) -> Vec<(usize, Option<usize>, f64)> {
        ocr_items
            .iter()
            .enumerate()
            .map(|(i, ocr_item)| {
                let table_strings: Vec<String> = table_items
                    .iter()
                    .map(|s| s.as_ref().to_string())
                    .collect();
                
                match self.match_order_number(ocr_item, &table_strings) {
                    Some((matched, score)) => {
                        let matched_idx = table_strings.iter().position(|s| s == &matched);
                        (i, matched_idx, score)
                    }
                    None => (i, None, 0.0),
                }
            })
            .collect()
    }
}
```

```rust
pub struct ExportService;

    pub fn export_errors_csv(
        errors: &[(String, String)],
        output_path: &str,
    ) -> Result<String, String> {
        let timestamp = Local::now().format("%Y-%m-%d %H:%M:%S");
        let filename = format!("错误列表_{}.csv", Local::now().format("%Y%m%d"));
        let full_path = if output_path.is_empty() {
            filename
        } else {
            format!("{}/{}", output_path, filename)
        };
        
        let mut file = File::create(&full_path)
            .map_err(|e| format!("创建文件失败: {}", e))?;
        
        // BOM for UTF-8
        file.write_all(b"\xef\xbb\xbf")
            .map_err(|e| format!("写入BOM失败: {}", e))?;
        
        // CSV Header
        writeln!(file, "文件名,错误信息,时间")
            .map_err(|e| format!("写入表头失败: {}", e))?;
        
        // CSV Content
        for (file_name, error_msg) in errors {
            writeln!(file, "{},{},{}", file_name, error_msg, timestamp)
                .map_err(|e| format!("写入数据失败: {}", e))?;
        }
        
        Ok(full_path)
    }

                record.reason
            ).map_err(|e| format!("写入数据失败: {}", e))?;
        }
        
        Ok(full_path)
    }
}

pub struct UnmatchedRecord {
    pub order_number: String,
    pub source_file: String,
    pub reason: String,
}
```
